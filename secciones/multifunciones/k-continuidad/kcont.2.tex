\begin{frame}
  \Thm{*}{
    Supongamos que $0\in K$ y sea $F:H\to\P_0(Y)$ una multifunción tal que
    para todo $x\in H$ el conjunto $F(x)\subseteq Y$ es acotado. Si $F$ es 
    $K$-midconvexa y $K$-acotada superior en un subconjunto $D\subseteq H$ con
    interior no vacío, entonces, $F$ es $K$-continua y $K$-convexa en $H$.
  }{\cite{Nik86}}

  \Thm{*}{
    Supongamos que $0\in K$ y sea $F:H\to\P_0(Y)$ una multifunción tal que
    para todo $x\in H$ el conjunto $F(x)\subseteq Y$ es cerrado y acotado. 
    Si $F$ es $K$-midcóncava y $K$-acotada inferior en un subconjunto $D\subseteq H$ con
    interior no vacío, entonces, $F$ es $K$-continua y $K$-cóncava en $H$.
 }{\cite{Nik86}}	
\end{frame}
