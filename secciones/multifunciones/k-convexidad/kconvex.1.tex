\begin{frame}{$K$-convexidad}
  Sea $K\subseteq Y$ un cono convexo. 
  Denotemos por $\U(X)$ y $\U(Y)$
  a las bases locales de $X$ y $Y$ respectivamente.
  \Defi{*}{
    Se dice que la multifunción $F$ es $K$-convexa 
    en $H$ si para todo $x,y\in H$ y para todo
    $t\in[0,1]$ se tiene que 
    \Eq{Kcvxsvm}{
      tF(x)+(1-t)F(y)\subseteq F(tx+(1-t)y)+K
    }
  }
  \Defi{*}{
    Se dice que la multifunción $F$ es $K$-cóncava 
    en $H$ si para todo $x,y\in H$  y para todo 
    $t\in[0,1]$ se tiene que 
    \Eq{Kccvsvm}{
      F(tx+(1-t)y)\subseteq tF(x)+(1-t)F(y)+K
    }
  }
\end{frame}
