\begin{frame}{Una equivalencia.}
  \Prp{*}{
    Sea $H\subseteq\R^n$ un conjunto convexo y 
    $f: H\subseteq\R^n\to\R$ una funci\'on. 
    $f$ es \emph{convexa} en $H$ sii, para todo $x,y\in H$ 
    se tiene que 
    \Eq{ytr}{
      f(tx + (1-t)y) \leq tf(x) + (1-t)f(y), \quad t\in[0,1].
    }
  }
\end{frame}
